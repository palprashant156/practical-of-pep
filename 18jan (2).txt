 #include <iostream>

using namespace std;

class Triangle
{
private:
    int x, y, z,ar;

public:
    Triangle(int a, int b, int c)
    {
        x = a;
        y = b;
        z = c;
    }
    int Area()
    {
        ar = 0.5*x*y;
        return ar;
    }
    int perimeter()
    {
        return x+y+z;
    }
};

int main()
{
    Triangle obj = Triangle(3,4,5);
    cout << obj.Area() <<" Units"<< "\n";
    cout<<obj.perimeter()<<" Units"<<"\n";
    return 0;
}
